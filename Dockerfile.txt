# Use Python as the base image
FROM python:3.9

# Install dependencies
RUN apt-get update && apt-get install -y wget curl unzip gnupg

# Install Google Chrome
RUN wget -q -O - https://dl.google.com/linux/linux_signing_key.pub | apt-key add - && \
    echo 'deb [arch=amd64] http://dl.google.com/linux/chrome/deb/ stable main' > /etc/apt/sources.list.d/google.list && \
    apt-get update && apt-get install -y google-chrome-stable

# Create a symlink so that Chrome is available at /usr/bin/google-chrome
RUN ln -sf /usr/bin/google-chrome-stable /usr/bin/google-chrome

# Verify installation (outputs chrome version)
RUN which google-chrome && google-chrome --version

# Set the working directory
WORKDIR /app

# Copy all project files into the container
COPY . .

# Upgrade pip and install Python dependencies
RUN pip install --upgrade pip && pip install -r requirements.txt

# Expose the Flask port
EXPOSE 5000

# Start the server using gunicorn
CMD ["gunicorn", "-b", "0.0.0.0:5000", "server:app"]
